#Transform excel file into pandas dataframe. Ensure first sheet in excel file has data, first row contains column headings
def excel_to_dataframe(datafile):
    workbook = xlrd.open_workbook(datafile)
    sheet = workbook.sheet_by_index(0)
    #Ensure first sheet has the data
    data = [[sheet.cell_value(r, col)
                for col in range(sheet.ncols)]
                    for r in range(sheet.nrows)]
    headers = data.pop(0)
    df = pd.DataFrame(data, columns = headers)
    return df

California Datasets¶

Public Schools: contains all public schools, CDS code, grades served. Year: 2016, updated daily. Source: http://www.cde.ca.gov/ds/si/ds/pubschls.asp

Enrollment: enrollment by grade, disaggregated by ehtnicity and gender. Years: 11/12, 12/13, 13/14, 14/15, 15/16. Source: http://www.cde.ca.gov/ds/sd/sd/filesenr.asp

API: School results on API (more details below table), test discontinued after 2014. Years: 3yr weighted average 11-14; 13 [Not including more years because it's old]. Source: http://www.cde.ca.gov/ta/ac/ap/apidatafiles.asp#updates

AYP: School results on AYP - % ELA and Math proficient, graduation rates, reporting discontinued after 2015. Years: 2013, 2014, 2015. Source: http://www.cde.ca.gov/ta/ac/ay/aypdatafiles.asp [Note: attendance rate is reported under AYP but only for K-8 grades; can be used to compare middle school performance.]

Graduates: Number of graduates from 12th grade, number of graduates meeting UC/CSU standards. Use for graduation rates when comparing enrollment for same year. Years: 10/11, 11/12, 12/13, 13/14, 14/15. Source: http://www.cde.ca.gov/ds/sd/sd/filesgrads.asp

Truancy: Number of truants per school & rates of truancy, not separated by grade. Years: 12/13, 13/14, 14/15. Source: http://www.cde.ca.gov/ds/sd/sd/filestd.asp Note: Census Enrollment: The total unique or unduplicated number of primary enrollments on the CALPADS Fall 1 Census Day, which is the first Wednesday in October of the associated academic year. Cumulative Enrollment: The total number of unique or unduplicated primary, secondary, and short-term enrollments within the academic year (July 1 to June 30), regardless of whether the student is enrolled multiple times within a school or district.

Suspension/Expulsion: Reported cases of suspensions and expulsions, separate row for each instance. Years: 11/12, 12/13, 13/14, 14/15. Source: http://www.cde.ca.gov/ds/sd/sd/filesesd.asp

## Functions to create grouped datasets and plots

# Create subset of dataset by school year
def subsetbyyear(df, year):
    newdf = df.loc[df.year == year]
    return newdf

# Create weighted averages for wavg columns - if there are no values for a year in one column, return Nan
def wavg(df, data_col, weight_col):
    try:
        d = df[data_col]
        w = df[weight_col]
        return (d * w).sum() / w.sum()
    except ZeroDivisionError:
        return float('nan')
#https://docs.python.org/2/tutorial/errors.html

# Create weighted average datasets   
def create_wavg_dataset(df, year):
    df1 = subsetbyyear(df, year)

    grouped = df1.groupby(df1.School_Category, as_index = False)

    ## ADD TO THIS WHEN ADDING ADDITIONAL DATA COLUMNS 
    new_df = grouped['high_tot'].mean()
    new_df['Year'] = df1.iloc[0,5]
    new_df['Graduate_rate'] = grouped.apply(wavg, 'gradrate', 'GR_12')
    new_df['UC_rate'] = grouped.apply(wavg, 'UC_rate', 'GRADS')
    new_df['Truancy Rate'] = grouped.apply(wavg, 'Truancy Rate', 'ENR_TOTAL')
    new_df['e_pprof'] = grouped.apply(wavg, 'e_pprof', 'e_val')
    new_df['m_pprof'] = grouped.apply(wavg, 'm_pprof', 'm_val')
    new_df['st_rank'] = grouped.apply(wavg, 'st_rank', 'ENR_TOTAL')
    new_df['sim_rank'] = grouped.apply(wavg, 'sim_rank', 'ENR_TOTAL')

    # Rename Other Schools

    # Reorder columns

def wavg(df, data_col, weight_col):
    try:
        d = df[data_col]
        w = df[weight_col]
        return (d * w).sum() / w.sum()
    except ZeroDivisionError:
        return float('nan')

def wavg(df, data_col, weight_col):
    try:
        d = df[data_col]
        w = df[weight_col]
        return (d * w).sum() / w.sum()
    except ZeroDivisionError:
        return float('nan')

#Formulas to create tables of schools by grade range and school type
def getgradedata(df, low_grade):
    new_df = df.loc[df['Low Grade'] == low_grade]
    return new_df

def groupbytype(df, colname):
    group = df.groupby(df['Type'], as_index = False)
    output = group['Sch ID'].count()
    output.rename(columns ={'Sch ID': colname,}, inplace=True)
    return output

def addgradescolumn(master_df, grades_df, colname):
    group = groupbytype(grades_df, colname)
    output = master_df.merge(group, how = 'outer', on = 'Type')
    return output

def apply3inputfunction(list_o_lists, function):
    for i in list_o_lists:
        function(i[0], i[1], i[2])
    return

def addpercentcolumn(df, coltodivide, colname):
    df[colname] = df[coltodivide]/df['Total Schools*']
    df[colname] = pd.Series(["{0:.2f}%".format(val * 100) for val in df[colname]], index = df.index)
    return
